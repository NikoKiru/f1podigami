`observe_order` math (exact; sign flips gradient only):

```python
def observe_order(self, entries, *, depth=None, weight=1.0, sign=1):
    m = len(entries)
    if m < 2 or weight <= 0.0:
        return
    r = m - 1 if depth is None else max(1, min(depth, m - 1))
    states = [(self.driver(d), self.constructor(c)) for d, c in entries]
    s = [max(-30.0, min(30.0, sign * (ds.mu + cs.mu))) for ds, cs in states]
    w = [math.exp(v) for v in s]
    Z = [0.0] * (m + 1)
    for j in range(m - 1, -1, -1):
        Z[j] = Z[j + 1] + w[j]
    pref1 = pref2 = 0.0
    grads, hesss = [], []
    for i in range(m):
        if i < r:
            inv = 1.0 / Z[i]
            pref1 += inv
            pref2 += inv * inv
        g = (1.0 if i < r else 0.0) - w[i] * pref1
        h = w[i] * pref1 - w[i] * w[i] * pref2
        grads.append(sign * g)
        hesss.append(max(h, 0.0))
    # per-constructor accumulation, then scalar Kalman steps
    con_g, con_h = {}, {}
    for (ds, cs), g, h in zip(states, grads, hesss, strict=True):
        self._nudge(ds, g, h, weight)
        key = id(cs)
        con_g[key] = (cs, con_g.get(key, (cs, 0.0))[1] + g)  # see impl note
    ...
def _nudge(self, st, g, h, weight):
    denom = 1.0 + st.var * weight * h
    st.mu += st.var * weight * g / denom
    st.var = min(25.0, max(1e-6, st.var / denom))
```

(Implementation note: collect constructor g/h sums in dicts keyed by lineage root id, then `_nudge` each once — a constructor must get ONE step with summed g/h, not one per driver.)

- [ ] Failing tests (hand-checkable):
  - two drivers, equal priors, A beats B → μ_A > μ_B, both var shrink; magnitudes symmetric.
  - `sign=-1` with A listed first → μ_A *decreases* (attrition: first picked = first failure).
  - `depth=1`, three entries → entry 3 gets the same grad/hess as entry 2 pattern (no stage-2 term): assert grad_3 == −w3/Z0-only (exposed via state deltas).
  - `weight=0.5` halves the step direction consistently (compare two engines).
  - teammates: both drive constructor X, finish 1st and last of 4 → constructor mean moves by the *summed* gradient once (compare vs manual expected).
  - `advance_season(2022)` (reg year) inflates constructor var by `season_var_con + reg_var_con`, driver var by `season_var_drv`; non-reg year only base terms; `advance_race` adds τ².

```python
def predict_race(entrants, mu_var, p_fin, temp, params, seen, *,
                 n_draws=512, seed=20260704, screen=250):
    rng = random.Random(seed)
    ids = sorted(entrants)
    seen_here = {t for t in seen if all(d in set(ids) for d in t)}
    lam0 = {d: math.exp(max(-30.0, min(30.0, mu_var[d][0]))) for d in ids}
    base = model.all_set_probs(lam0, ids)                    # screening only
    screened = [t for t, _ in sorted(base.items(), key=lambda kv: -kv[1])
                if t not in seen_here][:screen]
    track = {t: 0.0 for t in screened} | {t: 0.0 for t in seen_here}
    used = 0
    for _ in range(n_draws):
        s = {d: mu_var[d][0] + rng.gauss(0.0, 1.0) * math.sqrt(mu_var[d][1]) for d in ids}
        alive = [d for d in ids if rng.random() < p_fin[d]]
        t_eff = temp * (params["t_wild"] if rng.random() < params["p_wild"] else 1.0)
        if len(alive) < 3:
            continue
        used += 1
        w = {d: math.exp(max(-30.0, min(30.0, s[d] / t_eff))) for d in alive}
        tot = sum(w.values())
        aset = set(alive)
        for t in track:
            if t[0] in aset and t[1] in aset and t[2] in aset:
                track[t] += model.pl_set_prob(t, w, tot)
    probs = {t: v / max(used, 1) for t, v in track.items()}
    p_new = 1.0 - sum(probs[t] for t in seen_here)
    ranked_new = sorted(((t, p) for t, p in probs.items() if t not in seen_here),
                        key=lambda kv: (-kv[1], kv[0]))
    return {"trio_probs": probs, "p_new": p_new, "ranked_new": ranked_new, "draws_used": used}
```
